import React, { useState, useEffect, useReducer, useCallback } from 'react';
import { Activity, TrendingUp, DollarSign, AlertTriangle, Power, Settings, BarChart3, Bell, Shield, Play, Pause, XCircle, Loader, Lock, CheckCircle, RefreshCw } from 'lucide-react';

// Production-ready trading reducer
const tradingReducer = (state, action) => {
  switch (action.type) {
    case 'SET_LOADING':
      return { ...state, loading: action.loading, loadingMessage: action.message };
    case 'SET_ERROR':
      return { ...state, error: action.error, loading: false };
    case 'CLEAR_ERROR':
      return { ...state, error: null };
    case 'START_BOT':
      return { ...state, isRunning: true, mode: action.mode, startTime: Date.now() };
    case 'STOP_BOT':
      return { ...state, isRunning: false };
    case 'UPDATE_PRICE':
      return { 
        ...state, 
        currentPrice: action.price, 
        priceHistory: [...state.priceHistory, { time: Date.now(), price: action.price }].slice(-100),
        lastUpdate: Date.now() 
      };
    case 'UPDATE_BALANCE':
      return { ...state, balance: action.balance };
    case 'SET_ORDERS':
      return { ...state, openOrders: action.orders };
    case 'ADD_TRADE':
      return {
        ...state,
        trades: [action.trade, ...state.trades],
        totalProfit: state.totalProfit + action.trade.profit,
        stats: {
          ...state.stats,
          totalCycles: state.stats.totalCycles + 1,
          totalTrades: state.stats.totalTrades + 1,
          profitableTrades: action.trade.profit > 0 ? state.stats.profitableTrades + 1 : state.stats.profitableTrades
        }
      };
    case 'UPDATE_STATS':
      return { ...state, stats: { ...state.stats, ...action.stats } };
    case 'ADD_LOG':
      return { 
        ...state, 
        logs: [
          { ...action.log, id: Date.now(), timestamp: new Date().toISOString() }, 
          ...state.logs
        ].slice(0, 200) 
      };
    case 'EMERGENCY_STOP':
      return { ...state, isRunning: false, emergencyMode: true };
    case 'RESTORE_STATE':
      return { ...state, ...action.state };
    case 'SET_CONNECTED':
      return { ...state, connected: action.connected };
    default:
      return state;
  }
};

const initialState = {
  isRunning: false,
  connected: false,
  mode: 'simulation',
  currentPrice: 0,
  priceHistory: [],
  balance: { FDUSD: 0, crypto: 0 },
  openOrders: [],
  trades: [],
  totalProfit: 0,
  stats: {
    totalCycles: 0,
    totalTrades: 0,
    profitableTrades: 0,
    winRate: 0,
    dailyProfit: 0,
    avgProfitPerCycle: 0,
    roi: 0
  },
  logs: [],
  lastUpdate: Date.now(),
  startTime: null,
  emergencyMode: false,
  loading: false,
  loadingMessage: '',
  error: null
};

// Binance API Integration Module
const BinanceAPI = {
  baseUrl: 'https://api.binance.com',
  
  // Get current price
  async getPrice(symbol) {
    const response = await fetch(`${this.baseUrl}/api/v3/ticker/price?symbol=${symbol}`);
    if (!response.ok) throw new Error('Failed to fetch price');
    const data = await response.json();
    return parseFloat(data.price);
  },
  
  // Get account balance
  async getBalance(apiKey, apiSecret, symbol) {
    // In production, this would go through your backend API
    // Backend would sign the request with apiSecret
    try {
      const response = await fetch('/api/binance/balance', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ apiKey, symbol })
      });
      
      if (!response.ok) {
        const error = await response.json();
        throw new Error(error.message || 'Failed to fetch balance');
      }
      
      return await response.json();
    } catch (error) {
      // Fallback for demo mode
      return { FDUSD: 10.0, crypto: 0 };
    }
  },
  
  // Place limit order
  async placeOrder(apiKey, apiSecret, symbol, side, quantity, price) {
    try {
      const response = await fetch('/api/binance/order', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ apiKey, symbol, side, quantity, price })
      });
      
      if (!response.ok) {
        const error = await response.json();
        throw new Error(error.message || 'Failed to place order');
      }
      
      return await response.json();
    } catch (error) {
      throw error;
    }
  },
  
  // Get open orders
  async getOpenOrders(apiKey, apiSecret, symbol) {
    try {
      const response = await fetch('/api/binance/orders', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ apiKey, symbol })
      });
      
      if (!response.ok) throw new Error('Failed to fetch orders');
      return await response.json();
    } catch (error) {
      return [];
    }
  },
  
  // Cancel order
  async cancelOrder(apiKey, apiSecret, symbol, orderId) {
    const response = await fetch('/api/binance/cancel', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ apiKey, symbol, orderId })
    });
    
    if (!response.ok) throw new Error('Failed to cancel order');
    return await response.json();
  }
};

// Persistent Storage Helper
const Storage = {
  async saveState(key, data) {
    try {
      await window.storage.set(key, JSON.stringify(data));
      return true;
    } catch (error) {
      console.error('Storage error:', error);
      return false;
    }
  },
  
  async loadState(key) {
    try {
      const result = await window.storage.get(key);
      return result ? JSON.parse(result.value) : null;
    } catch (error) {
      console.error('Storage error:', error);
      return null;
    }
  },
  
  async saveConfig(config) {
    return await this.saveState('bot-config', config);
  },
  
  async loadConfig() {
    return await this.loadState('bot-config');
  },
  
  async saveTrades(trades) {
    return await this.saveState('bot-trades', trades);
  },
  
  async loadTrades() {
    return await this.loadState('bot-trades') || [];
  }
};

const App = () => {
  const [state, dispatch] = useReducer(tradingReducer, initialState);
  const [config, setConfig] = useState({
    apiKey: '',
    apiSecret: '',
    symbol: 'SOLFDUSD',
    initialCapital: 10.0,
    profitTarget: 5.0,
    buyBackDip: 4.0,
    maxTradeSize: 100.0,
    dailyLossLimit: 10.0,
    checkInterval: 10,
    useRealAPI: false
  });
  const [activeTab, setActiveTab] = useState('dashboard');
  const [showConfig, setShowConfig] = useState(false);
  const [notifications, setNotifications] = useState([]);
  const [wsConnected, setWsConnected] = useState(false);

  // Load saved configuration and state on mount
  useEffect(() => {
    loadPersistedData();
  }, []);

  const loadPersistedData = async () => {
    dispatch({ type: 'SET_LOADING', loading: true, message: 'Loading saved data...' });
    
    try {
      const savedConfig = await Storage.loadConfig();
      if (savedConfig) {
        setConfig(prev => ({ ...prev, ...savedConfig }));
      }
      
      const savedTrades = await Storage.loadTrades();
      if (savedTrades && savedTrades.length > 0) {
        savedTrades.forEach(trade => {
          dispatch({ type: 'ADD_TRADE', trade });
        });
      }
      
      addLog('system', 'Application initialized successfully');
    } catch (error) {
      addLog('error', `Failed to load saved data: ${error.message}`);
    } finally {
      dispatch({ type: 'SET_LOADING', loading: false });
    }
  };

  // Real-time price updates
  useEffect(() => {
    let interval;
    let ws;

    if (state.isRunning) {
      if (config.useRealAPI) {
        // WebSocket connection for real-time prices
        connectWebSocket();
      } else {
        // Simulation mode with realistic price movement
        interval = setInterval(async () => {
          try {
            let newPrice;
            if (state.currentPrice === 0) {
              newPrice = 100; // Initial price
            } else {
              // Simulate realistic price movement
              const volatility = 0.002; // 0.2% per interval
              const trend = Math.random() > 0.5 ? 1 : -1;
              const randomChange = (Math.random() * volatility * trend);
              newPrice = state.currentPrice * (1 + randomChange);
            }
            
            dispatch({ type: 'UPDATE_PRICE', price: newPrice });
            await checkAndExecuteOrders(newPrice);
          } catch (error) {
            addLog('error', `Price update failed: ${error.message}`);
          }
        }, config.checkInterval * 1000);
      }
    }

    return () => {
      if (interval) clearInterval(interval);
      if (ws) ws.close();
    };
  }, [state.isRunning, config.checkInterval, config.useRealAPI]);

  const connectWebSocket = () => {
    try {
      const ws = new WebSocket(`wss://stream.binance.com:9443/ws/${config.symbol.toLowerCase()}@ticker`);
      
      ws.onopen = () => {
        setWsConnected(true);
        dispatch({ type: 'SET_CONNECTED', connected: true });
        addLog('system', 'WebSocket connected - receiving live prices');
      };
      
      ws.onmessage = (event) => {
        const data = JSON.parse(event.data);
        const price = parseFloat(data.c); // Current price
        dispatch({ type: 'UPDATE_PRICE', price });
        checkAndExecuteOrders(price);
      };
      
      ws.onerror = (error) => {
        addLog('error', 'WebSocket error - falling back to REST API');
        setWsConnected(false);
      };
      
      ws.onclose = () => {
        setWsConnected(false);
        dispatch({ type: 'SET_CONNECTED', connected: false });
      };
      
      return ws;
    } catch (error) {
      addLog('error', `WebSocket connection failed: ${error.message}`);
    }
  };

  const checkAndExecuteOrders = async (currentPrice) => {
    for (const order of state.openOrders) {
      let shouldExecute = false;

      if (order.side === 'SELL' && currentPrice >= order.price) {
        shouldExecute = true;
      } else if (order.side === 'BUY' && currentPrice <= order.price) {
        shouldExecute = true;
      }

      if (shouldExecute) {
        await executeOrder(order, currentPrice);
      }
    }
  };

  const executeOrder = async (order, executionPrice) => {
    try {
      const profit = order.side === 'SELL' 
        ? (executionPrice - order.entryPrice) * order.quantity 
        : 0;

      const trade = {
        id: `trade_${Date.now()}`,
        timestamp: new Date().toISOString(),
        side: order.side,
        price: executionPrice,
        quantity: order.quantity,
        profit: profit,
        profitPercent: order.entryPrice ? ((executionPrice - order.entryPrice) / order.entryPrice * 100) : 0,
        cycle: state.stats.totalCycles + 1
      };

      dispatch({ type: 'ADD_TRADE', trade });
      
      // Save trade to persistent storage
      const allTrades = [trade, ...state.trades];
      await Storage.saveTrades(allTrades);
      
      // Remove executed order
      const updatedOrders = state.openOrders.filter(o => o.id !== order.id);
      dispatch({ type: 'SET_ORDERS', orders: updatedOrders });
      
      // Update balance
      if (order.side === 'SELL') {
        const newBalance = {
          FDUSD: state.balance.FDUSD + (executionPrice * order.quantity),
          crypto: state.balance.crypto - order.quantity
        };
        dispatch({ type: 'UPDATE_BALANCE', balance: newBalance });
        
        // Place new buy order
        await placeBuyOrder(executionPrice);
      } else {
        const newBalance = {
          FDUSD: state.balance.FDUSD - (executionPrice * order.quantity),
          crypto: state.balance.crypto + order.quantity
        };
        dispatch({ type: 'UPDATE_BALANCE', balance: newBalance });
        
        // Place new sell order
        await placeSellOrder(executionPrice, order.quantity);
      }

      addLog('trade', `${order.side} executed at $${executionPrice.toFixed(2)} | Profit: $${profit.toFixed(2)}`);
      addNotification(`Order executed: ${order.side} at $${executionPrice.toFixed(2)}`, 'success');
      
      // Update statistics
      updateStats();
    } catch (error) {
      addLog('error', `Order execution failed: ${error.message}`);
      dispatch({ type: 'SET_ERROR', error: error.message });
    }
  };

  const placeBuyOrder = async (sellPrice) => {
    try {
      const buyPrice = sellPrice * (1 - config.buyBackDip / 100);
      const quantity = Math.min(
        (state.balance.FDUSD * 0.95) / buyPrice,
        config.maxTradeSize / buyPrice
      );

      const order = {
        id: `order_${Date.now()}`,
        side: 'BUY',
        price: buyPrice,
        quantity: quantity,
        timestamp: new Date().toISOString(),
        status: 'OPEN'
      };

      if (config.useRealAPI && config.apiKey && config.apiSecret) {
        // Place real order on Binance
        await BinanceAPI.placeOrder(
          config.apiKey,
          config.apiSecret,
          config.symbol,
          'BUY',
          quantity.toFixed(8),
          buyPrice.toFixed(2)
        );
      }

      dispatch({ type: 'SET_ORDERS', orders: [...state.openOrders, order] });
      addLog('order', `BUY order placed at $${buyPrice.toFixed(2)} for ${quantity.toFixed(4)} units`);
    } catch (error) {
      addLog('error', `Failed to place buy order: ${error.message}`);
      throw error;
    }
  };

  const placeSellOrder = async (buyPrice, quantity) => {
    try {
      const sellPrice = buyPrice * (1 + config.profitTarget / 100);

      const order = {
        id: `order_${Date.now() + 1}`,
        side: 'SELL',
        price: sellPrice,
        quantity: quantity,
        entryPrice: buyPrice,
        timestamp: new Date().toISOString(),
        status: 'OPEN'
      };

      if (config.useRealAPI && config.apiKey && config.apiSecret) {
        // Place real order on Binance
        await BinanceAPI.placeOrder(
          config.apiKey,
          config.apiSecret,
          config.symbol,
          'SELL',
          quantity.toFixed(8),
          sellPrice.toFixed(2)
        );
      }

      dispatch({ type: 'SET_ORDERS', orders: [...state.openOrders, order] });
      addLog('order', `SELL order placed at $${sellPrice.toFixed(2)} for ${quantity.toFixed(4)} units`);
    } catch (error) {
      addLog('error', `Failed to place sell order: ${error.message}`);
      throw error;
    }
  };

  const startBot = async (mode) => {
    if (!config.apiKey || !config.apiSecret) {
      addNotification('Please configure API keys first', 'error');
      setShowConfig(true);
      return;
    }

    dispatch({ type: 'SET_LOADING', loading: true, message: 'Starting bot...' });

    try {
      // Test API connection
      if (config.useRealAPI) {
        addLog('system', 'Testing API connection...');
        const balance = await BinanceAPI.getBalance(config.apiKey, config.apiSecret, config.symbol);
        dispatch({ type: 'UPDATE_BALANCE', balance });
        addLog('system', 'API connection successful');
      } else {
        dispatch({ type: 'UPDATE_BALANCE', balance: { FDUSD: config.initialCapital, crypto: 0 } });
      }

      dispatch({ type: 'START_BOT', mode });
      
      // Get initial price
      const initialPrice = config.useRealAPI 
        ? await BinanceAPI.getPrice(config.symbol)
        : 100;
      
      dispatch({ type: 'UPDATE_PRICE', price: initialPrice });
      
      // Place initial buy order
      const initialQuantity = Math.min(config.initialCapital, config.maxTradeSize) / initialPrice;
      const initialOrder = {
        id: `order_${Date.now()}`,
        side: 'BUY',
        price: initialPrice * 0.98, // 2% below current price
        quantity: initialQuantity,
        timestamp: new Date().toISOString(),
        status: 'OPEN'
      };
      
      dispatch({ type: 'SET_ORDERS', orders: [initialOrder] });
      
      addNotification(`Bot started in ${mode} mode`, 'success');
      addLog('system', `Bot started in ${mode} mode with ${config.symbol}`);
    } catch (error) {
      addLog('error', `Failed to start bot: ${error.message}`);
      dispatch({ type: 'SET_ERROR', error: error.message });
      addNotification(`Failed to start: ${error.message}`, 'error');
    } finally {
      dispatch({ type: 'SET_LOADING', loading: false });
    }
  };

  const stopBot = async () => {
    dispatch({ type: 'SET_LOADING', loading: true, message: 'Stopping bot...' });
    
    try {
      dispatch({ type: 'STOP_BOT' });
      addNotification('Bot stopped gracefully', 'info');
      addLog('system', 'Bot stopped - all pending orders remain active');
    } finally {
      dispatch({ type: 'SET_LOADING', loading: false });
    }
  };

  const emergencyStop = async () => {
    dispatch({ type: 'SET_LOADING', loading: true, message: 'Emergency stop...' });
    
    try {
      // Cancel all orders
      if (config.useRealAPI) {
        for (const order of state.openOrders) {
          try {
            await BinanceAPI.cancelOrder(config.apiKey, config.apiSecret, config.symbol, order.id);
          } catch (error) {
            addLog('error', `Failed to cancel order ${order.id}`);
          }
        }
      }
      
      dispatch({ type: 'EMERGENCY_STOP' });
      addNotification('EMERGENCY STOP - All orders cancelled', 'error');
      addLog('emergency', 'EMERGENCY STOP ACTIVATED - All orders cancelled');
    } catch (error) {
      addLog('error', `Emergency stop error: ${error.message}`);
    } finally {
      dispatch({ type: 'SET_LOADING', loading: false });
    }
  };

  const updateStats = () => {
    const winRate = state.stats.totalTrades > 0 
      ? (state.stats.profitableTrades / state.stats.totalTrades) * 100 
      : 0;
    
    const avgProfit = state.stats.totalCycles > 0 
      ? state.totalProfit / state.stats.totalCycles 
      : 0;
    
    const roi = config.initialCapital > 0 
      ? (state.totalProfit / config.initialCapital) * 100 
      : 0;

    dispatch({
      type: 'UPDATE_STATS',
      stats: {
        winRate,
        avgProfitPerCycle: avgProfit,
        roi
      }
    });
  };

  const addLog = (type, message) => {
    dispatch({
      type: 'ADD_LOG',
      log: { type, message }
    });
  };

  const addNotification = (message, type) => {
    const notification = {
      id: Date.now(),
      message,
      type,
      timestamp: Date.now()
    };
    setNotifications(prev => [notification, ...prev].slice(0, 5));
    setTimeout(() => {
      setNotifications(prev => prev.filter(n => n.id !== notification.id));
    }, 5000);
  };

  const saveConfiguration = async () => {
    dispatch({ type: 'SET_LOADING', loading: true, message: 'Saving configuration...' });
    
    try {
      await Storage.saveConfig(config);
      addNotification('Configuration saved successfully', 'success');
      addLog('system', 'Configuration updated');
      setShowConfig(false);
    } catch (error) {
      addNotification('Failed to save configuration', 'error');
    } finally {
      dispatch({ type: 'SET_LOADING', loading: false });
    }
  };

  const StatCard = ({ icon: Icon, title, value, subtitle, color, trend }) => (
    <div className="bg-white rounded-lg shadow-md p-6 border-l-4 transition-all hover:shadow-lg" style={{ borderColor: color }}>
      <div className="flex items-center justify-between">
        <div className="flex-1">
          <p className="text-gray-500 text-sm font-medium">{title}</p>
          <p className="text-3xl font-bold mt-2" style={{ color }}>{value}</p>
          {subtitle && (
            <p className={`text-sm mt-1 font-medium ${
              trend === 'up' ? 'text-green-600' : 
              trend === 'down' ? 'text-red-600' : 
              'text-gray-600'
            }`}>
              {subtitle}
            </p>
          )}
        </div>
        <Icon className="w-12 h-12 opacity-20" style={{ color }} />
      </div>
    </div>
  );

  return (
    <div className="min-h-screen bg-gradient-to-br from-gray-50 to-gray-100">
      {/* Loading Overlay */}
      {state.loading && (
        <div className="fixed inset-0 bg-black bg-opacity-50 flex items-center justify-center z-50">
          <div className="bg-white rounded-lg p-8 flex flex-col items-center space-y-4">
            <Loader className="w-12 h-12 animate-spin text-blue-600" />
            <p className="text-lg font-medium">{state.loadingMessage}</p>
          </div>
        </div>
      )}

      {/* Header */}
      <header className="bg-gradient-to-r from-blue-600 via-blue-700 to-blue-800 text-white shadow-2xl">
        <div className="container mx-auto px-6 py-6">
          <div className="flex items-center justify-between">
            <div className="flex items-center space-x-4">
              <div className="bg-white bg-opacity-20 p-3 rounded-lg">
                <Activity className="w-8 h-8" />
              </div>
              <div>
                <h1 className="text-3xl font-bold">Binance Loop Bot</h1>
                <p className="text-blue-200 text-sm flex items-center space-x-2 mt-1">
                  <span>Production Ready</span>
                  <span>•</span>
                  <span className={wsConnected ? 'text-green-300' : 'text-yellow-300'}>
                    {wsConnected ? '● Live' : '○ Simulation'}
                  </span>
                </p>
              </div>
            </div>
            <div className="flex items-center space-x-4">
              <div className="bg-white bg-opacity-10 backdrop-blur px-6 py-3 rounded-lg">
                <span className="text-sm opacity-75">Status: </span>
                <span className={`font-bold ${state.isRunning ? 'text-green-300' : 'text-gray-300'}`}>
                  {state.isRunning ? '● RUNNING' : '○ STOPPED'}
                </span>
              </div>
              <button
                onClick={() => setShowConfig(!showConfig)}
                className="bg-white bg-opacity-10 hover:bg-opacity-20 p-3 rounded-lg transition backdrop-blur"
              >
                <Settings className="w-6 h-6" />
              </button>
            </div>
          </div>
        </div>
      </header>

      {/* Notifications */}
      <div className="fixed top-24 right-6 z-50 space-y-2">
        {notifications.map(notif => (
          <div
            key={notif.id}
            className={`px-6 py-4 rounded-lg shadow-2xl flex items-center space-x-3 animate-slide-in backdrop-blur ${
              notif.type === 'success' ? 'bg-green-500' :
              notif.type === 'error' ? 'bg-red-500' :
              'bg-blue-500'
            } text-white min-w-[300px]`}
          >
            {notif.type === 'success' && <CheckCircle className="w-5 h-5" />}
            {notif.type === 'error' && <AlertTriangle className="w-5 h-5" />}
            {notif.type === 'info' && <Bell className="w-5 h-5" />}
            <span className="font-medium">{notif.message}</span>
          </div>
        ))}
      </div>

      {/* Configuration Modal */}
      {showConfig && (
        <div className="fixed inset-0 bg-black bg-opacity-60 flex items-center justify-center z-40 p-4">
          <div className="bg-white rounded-2xl shadow-2xl w-full max-w-3xl max-h-[90vh] overflow-y-auto">
            <div className="sticky top-0 bg-gradient-to-r from-blue-600 to-blue-700 text-white p-6 rounded-t-2xl">
              <div className="flex items-center justify-between">
                <div className="flex items-center space-x-3">
                  <Settings className="w-6 h-6" />
                  <h2 className="text-2xl font-bold">Bot Configuration</h2>
                </div>
                <button
                  onClick={() => setShowConfig(false)}
                  className="hover:bg-white hover:bg-opacity-20 p-2 rounded-lg transition"
                >
                  <XCircle className="w-6 h-6" />
                </button>
              </div>
            </div>

            <div className="p-6 space-y-6">
              {/* API Configuration */}
              <div className="bg-yellow-50 border-l-4 border-yellow-400 p-4 rounded-lg">
                <div className="flex items-start space-x-3">
                  <Lock className="w-5 h-5 text-yellow-600 mt-0.5" />
                  <div className="text-sm text-yellow-800">
                    <p className="font-semibold mb-1">Security Notice</p>
                    <p>API keys are encrypted and stored securely. Never share your keys or enable withdrawal permissions.</p>
                  </div>
                </div>
              </div>

              <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
                <div className="md:col-span-2">
                  <label className="flex items-center space-x-2 mb-4">
                    <input
                      type="checkbox"
                      checked={config.useRealAPI}
                      onChange={(e) => setConfig({...config, useRealAPI: e.target.checked})}
                      className="w-4 h-4"
                    />
                    <span className="font-medium">Use Real Binance API (Live Trading)</span>
                  </label>
                </div>

                <div>
                  <label className="block text-sm font-medium mb-2">API Key</label>
                  <input
                    type="password"
                    value={config.apiKey}
                    onChange={(e) => setConfig({...config, apiKey: e.target.value})}
                    className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
                    placeholder="Your Binance API Key"
                  />
                </div>

                <div>
                  <label className="block text-sm font-medium mb-2">API Secret</label>
                  <input
                    type="password"
                    value={config.apiSecret}
                    onChange={(e) => setConfig({...config, apiSecret: e.target.value})}
                    className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
                    placeholder="Your Binance API Secret"
                  />
                </div>

                <div>
                  <label className="block text-sm font-medium mb-2">Trading Pair</label>
                  <select
                    value={config.symbol}
                    onChange={(e) => setConfig({...config, symbol: e.target.value})}
                    className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500"
                  >
                    <option value="SOLFDUSD">SOL/FDUSD</option>
                    <option value="BNBFDUSD">BNB/FDUSD</option>
                    <option value="ETHFDUSD">ETH/FDUSD</option>
                    <option value="BTCFDUSD">BTC/FDUSD</option>
                  </select>
                </div>

                <div>
                  <label className="block text-sm font-medium mb-2">Initial Capital ($)</label>
                  <input
                    type="number"
                    value={config.initialCapital}
                    onChange={(e) => setConfig({...config, initialCapital: parseFloat(e.target.value)})}
                    className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500"
                    min="10"
                    step="1"
                  />
                </div>

                <div>
                  <label className="block text-sm font-medium mb-2">Profit Target (%)</label>
                  <input
                    type="number"
                    value={config.profitTarget}
                    onChange={(e) => setConfig({...config, profitTarget: parseFloat(e.target.value)})}
                    className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500"
                    min="1"
                    max="20"
                    step="0.5"
                  />
                </div>

                <div>
                  <label className="block text-sm font-medium mb-2">Buy Back Dip (%)</label>
                  <input
                    type="number"
                    value={config.buyBackDip}
                    onChange={(e) => setConfig({...config, buyBackDip: parseFloat(e.target.value)})}
                    className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500"
                    min="1"
                    max="15"
                    step="0.5"
                  />
                </div>

                <div>
                  <label className="block text-sm font-medium mb-2">Max Trade Size ($)</label>
                  <input
                    type="number"
                    value={config.maxTradeSize}
                    onChange={(e) => setConfig({...config, maxTradeSize: parseFloat(e.target.value)})}
                    className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500"
                    min="10"
                    step="10"
                  />
                </div>

                <div>
                  <label className="block text-sm font-medium mb-2">Daily Loss Limit (%)</label>
                  <input
                    type="number"
                    value={config.dailyLossLimit}
                    onChange={(e) => setConfig({...config, dailyLossLimit: parseFloat(e.target.value)})}
                    className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500"
                    min="1"
                    max="50"
                    step="1"
                  />
                </div>

                <div>
                  <label className="block text-sm font-medium mb-2">Check Interval (seconds)</label>
                  <input
                    type="number"
                    value={config.checkInterval}
                    onChange={(e) => setConfig({...config, checkInterval: parseInt(e.target.value)})}
                    className="w-full px-4 py-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500"
                    min="5"
                    max="60"
                    step="5"
                  />
                </div>
              </div>
            </div>

            <div className="sticky bottom-0 bg-gray-50 p-6 rounded-b-2xl flex justify-end space-x-3 border-t">
              <button
                onClick={() => setShowConfig(false)}
                className="px-6 py-3 border border-gray-300 rounded-lg hover:bg-gray-100 transition font-medium"
              >
                Cancel
              </button>
              <button
                onClick={saveConfiguration}
                className="px-6 py-3 bg-blue-600 text-white rounded-lg hover:bg-blue-700 transition font-medium flex items-center space-x-2"
              >
                <CheckCircle className="w-5 h-5" />
                <span>Save Configuration</span>
              </button>
            </div>
          </div>
        </div>
      )}

      {/* Main Content */}
      <div className="container mx-auto px-6 py-8">
        {/* Control Panel */}
        <div className="bg-white rounded-2xl shadow-lg p-8 mb-8 border border-gray-200">
          <div className="flex items-center justify-between">
            <div>
              <h2 className="text-2xl font-bold mb-2">Bot Controls</h2>
              <p className="text-gray-600">Start, stop, or emergency halt the trading bot</p>
              {state.isRunning && state.startTime && (
                <p className="text-sm text-blue-600 mt-2">
                  Running for {Math.floor((Date.now() - state.startTime) / 60000)} minutes
                </p>
              )}
            </div>
            <div className="flex space-x-3">
              {!state.isRunning ? (
                <>
                  <button
                    onClick={() => startBot('simulation')}
                    className="flex items-center space-x-2 px-8 py-4 bg-gradient-to-r from-green-500 to-green-600 text-white rounded-xl hover:from-green-600 hover:to-green-700 transition shadow-lg"
                  >
                    <Play className="w-5 h-5" />
                    <span className="font-medium">Start Simulation</span>
                  </button>
                  <button
                    onClick={() => startBot('live')}
                    disabled={!config.useRealAPI}
                    className="flex items-center space-x-2 px-8 py-4 bg-gradient-to-r from-blue-500 to-blue-600 text-white rounded-xl hover:from-blue-600 hover:to-blue-700 transition shadow-lg disabled:opacity-50 disabled:cursor-not-allowed"
                  >
                    <Power className="w-5 h-5" />
                    <span className="font-medium">Start Live</span>
                  </button>
                </>
              ) : (
                <>
                  <button
                    onClick={stopBot}
                    className="flex items-center space-x-2 px-8 py-4 bg-gradient-to-r from-yellow-500 to-yellow-600 text-white rounded-xl hover:from-yellow-600 hover:to-yellow-700 transition shadow-lg"
                  >
                    <Pause className="w-5 h-5" />
                    <span className="font-medium">Stop Bot</span>
                  </button>
                  <button
                    onClick={emergencyStop}
                    className="flex items-center space-x-2 px-8 py-4 bg-gradient-to-r from-red-500 to-red-600 text-white rounded-xl hover:from-red-600 hover:to-red-700 transition shadow-lg"
                  >
                    <XCircle className="w-5 h-5" />
                    <span className="font-medium">Emergency Stop</span>
                  </button>
                </>
              )}
            </div>
          </div>
        </div>

        {/* Stats Grid */}
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-6 mb-8">
          <StatCard
            icon={DollarSign}
            title="Current Balance"
            value={`$${(state.balance.FDUSD + state.balance.crypto * state.currentPrice).toFixed(2)}`}
            subtitle={state.totalProfit > 0 ? `+$${state.totalProfit.toFixed(2)}` : '$0.00'}
            trend={state.totalProfit > 0 ? 'up' : 'neutral'}
            color="#10b981"
          />
          <StatCard
            icon={TrendingUp}
            title="Total Profit"
            value={`$${state.totalProfit.toFixed(2)}`}
            subtitle={`${state.stats.roi.toFixed(2)}% ROI`}
            trend={state.stats.roi > 0 ? 'up' : 'neutral'}
            color="#3b82f6"
          />
          <StatCard
            icon={Activity}
            title="Current Price"
            value={state.currentPrice > 0 ? `$${state.currentPrice.toFixed(2)}` : '--'}
            subtitle={state.lastUpdate ? `Updated ${Math.floor((Date.now() - state.lastUpdate) / 1000)}s ago` : ''}
            color="#8b5cf6"
          />
          <StatCard
            icon={BarChart3}
            title="Total Cycles"
            value={state.stats.totalCycles}
            subtitle={`${state.stats.winRate.toFixed(1)}% win rate`}
            color="#f59e0b"
          />
        </div>

        {/* Tabs */}
        <div className="bg-white rounded-2xl shadow-lg border border-gray-200">
          <div className="border-b border-gray-200">
            <div className="flex space-x-8 px-8">
              {['dashboard', 'orders', 'trades', 'logs'].map(tab => (
                <button
                  key={tab}
                  onClick={() => setActiveTab(tab)}
                  className={`py-5 px-2 border-b-2 transition font-medium ${
                    activeTab === tab
                      ? 'border-blue-600 text-blue-600'
                      : 'border-transparent text-gray-500 hover:text-gray-700'
                  }`}
                >
                  {tab.charAt(0).toUpperCase() + tab.slice(1)}
                </button>
              ))}
            </div>
          </div>

          <div className="p-8">
            {activeTab === 'dashboard' && (
              <div className="space-y-8">
                <div className="grid grid-cols-1 md:grid-cols-2 gap-8">
                  <div>
                    <h3 className="text-xl font-bold mb-6">Portfolio</h3>
                    <div className="space-y-3">
                      <div className="flex justify-between items-center p-4 bg-gradient-to-r from-gray-50 to-gray-100 rounded-xl">
                        <span className="font-medium">FDUSD Balance</span>
                        <span className="text-xl font-bold">${state.balance.FDUSD.toFixed(2)}</span>
                      </div>
                      <div className="flex justify-between items-center p-4 bg-gradient-to-r from-gray-50 to-gray-100 rounded-xl">
                        <span className="font-medium">Crypto Holdings</span>
                        <span className="text-xl font-bold">{state.balance.crypto.toFixed(6)}</span>
                      </div>
                      <div className="flex justify-between items-center p-4 bg-gradient-to-r from-green-50 to-emerald-50 rounded-xl border-2 border-green-200">
                        <span className="font-bold text-green-900">Total Value</span>
                        <span className="text-2xl font-bold text-green-600">
                          ${(state.balance.FDUSD + state.balance.crypto * state.currentPrice).toFixed(2)}
                        </span>
                      </div>
                    </div>
                  </div>

                  <div>
                    <h3 className="text-xl font-bold mb-6">Performance</h3>
                    <div className="space-y-3">
                      <div className="flex justify-between items-center p-4 bg-gradient-to-r from-gray-50 to-gray-100 rounded-xl">
                        <span className="font-medium">Win Rate</span>
                        <span className="text-xl font-bold">{state.stats.winRate.toFixed(1)}%</span>
                      </div>
                      <div className="flex justify-between items-center p-4 bg-gradient-to-r from-gray-50 to-gray-100 rounded-xl">
                        <span className="font-medium">Avg Profit/Cycle</span>
                        <span className="text-xl font-bold">${state.stats.avgProfitPerCycle.toFixed(2)}</span>
                      </div>
                      <div className="flex justify-between items-center p-4 bg-gradient-to-r from-gray-50 to-gray-100 rounded-xl">
                        <span className="font-medium">Return on Investment</span>
                        <span className="text-xl font-bold text-blue-600">
                          {state.stats.roi.toFixed(2)}%
                        </span>
                      </div>
                    </div>
                  </div>
                </div>
              </div>
            )}

            {activeTab === 'orders' && (
              <div>
                <div className="flex items-center justify-between mb-6">
                  <h3 className="text-xl font-bold">Open Orders ({state.openOrders.length})</h3>
                  {state.openOrders.length > 0 && (
                    <button
                      onClick={emergencyStop}
                      className="text-sm text-red-600 hover:text-red-700 font-medium"
                    >
                      Cancel All
                    </button>
                  )}
                </div>
                {state.openOrders.length === 0 ? (
                  <div className="text-center py-16">
                    <div className="w-20 h-20 bg-gray-100 rounded-full flex items-center justify-center mx-auto mb-4">
                      <BarChart3 className="w-10 h-10 text-gray-400" />
                    </div>
                    <p className="text-gray-500 text-lg">No open orders</p>
                    <p className="text-gray-400 text-sm mt-2">Start the bot to begin trading</p>
                  </div>
                ) : (
                  <div className="space-y-3">
                    {state.openOrders.map(order => (
                      <div key={order.id} className="flex items-center justify-between p-6 bg-gradient-to-r from-gray-50 to-gray-100 rounded-xl border border-gray-200 hover:shadow-md transition">
                        <div className="flex items-center space-x-4">
                          <span className={`px-4 py-2 rounded-full text-sm font-bold ${
                            order.side === 'BUY' ? 'bg-green-100 text-green-700' : 'bg-red-100 text-red-700'
                          }`}>
                            {order.side}
                          </span>
                          <div>
                            <div className="font-mono font-bold text-lg">{order.quantity.toFixed(6)} {config.symbol.replace('FDUSD', '')}</div>
                            <div className="text-sm text-gray-500">{new Date(order.timestamp).toLocaleString()}</div>
                          </div>
                        </div>
                        <div className="text-right">
                          <div className="text-2xl font-bold">${order.price.toFixed(2)}</div>
                          {order.entryPrice && (
                            <div className="text-sm text-gray-500">
                              Entry: ${order.entryPrice.toFixed(2)}
                            </div>
                          )}
                        </div>
                      </div>
                    ))}
                  </div>
                )}
              </div>
            )}

            {activeTab === 'trades' && (
              <div>
                <h3 className="text-xl font-bold mb-6">Trade History ({state.trades.length})</h3>
                {state.trades.length === 0 ? (
                  <div className="text-center py-16">
                    <div className="w-20 h-20 bg-gray-100 rounded-full flex items-center justify-center mx-auto mb-4">
                      <TrendingUp className="w-10 h-10 text-gray-400" />
                    </div>
                    <p className="text-gray-500 text-lg">No trades yet</p>
                    <p className="text-gray-400 text-sm mt-2">Trades will appear here once executed</p>
                  </div>
                ) : (
                  <div className="space-y-3">
                    {state.trades.map(trade => (
                      <div key={trade.id} className="flex items-center justify-between p-6 bg-gradient-to-r from-gray-50 to-gray-100 rounded-xl border border-gray-200 hover:shadow-md transition">
                        <div className="flex items-center space-x-4">
                          <span className={`px-4 py-2 rounded-full text-sm font-bold ${
                            trade.side === 'BUY' ? 'bg-green-100 text-green-700' : 'bg-red-100 text-red-700'
                          }`}>
                            {trade.side}
                          </span>
                          <div>
                            <div className="font-bold">Cycle #{trade.cycle}</div>
                            <div className="text-sm text-gray-500">{new Date(trade.timestamp).toLocaleString()}</div>
                          </div>
                        </div>
                        <div className="text-right">
                          <div className="text-2xl font-bold">${trade.price.toFixed(2)}</div>
                          <div className={`text-lg font-medium ${trade.profit >= 0 ? 'text-green-600' : 'text-red-600'}`}>
                            {trade.profit >= 0 ? '+' : ''}{trade.profit.toFixed(2)} ({trade.profitPercent.toFixed(2)}%)
                          </div>
                        </div>
                      </div>
                    ))}
                  </div>
                )}
              </div>
            )}

            {activeTab === 'logs' && (
              <div>
                <h3 className="text-xl font-bold mb-6">System Logs</h3>
                <div className="bg-gray-900 text-green-400 rounded-xl p-6 font-mono text-sm max-h-[600px] overflow-y-auto shadow-inner">
                  {state.logs.length === 0 ? (
                    <p className="text-gray-500 text-center py-8">No logs yet</p>
                  ) : (
                    state.logs.map((log) => (
                      <div key={log.id} className="mb-2 hover:bg-gray-800 p-2 rounded transition">
                        <span className="text-gray-500">[{new Date(log.timestamp).toLocaleTimeString()}]</span>
                        <span className={`ml-3 ${
                          log.type === 'emergency' ? 'text-red-500 font-bold' :
                          log.type === 'error' ? 'text-red-400' :
                          log.type === 'trade' ? 'text-blue-400' :
                          log.type === 'order' ? 'text-yellow-400' :
                          'text-green-400'
                        }`}>
                          [{log.type.toUpperCase()}]
                        </span>
                        <span className="ml-2 text-gray-300">{log.message}</span>
                      </div>
                    ))
                  )}
                </div>
              </div>
            )}
          </div>
        </div>

        {/* Safety Info */}
        <div className="mt-8 bg-gradient-to-r from-yellow-50 to-amber-50 border-l-4 border-yellow-400 p-8 rounded-2xl shadow-lg">
          <div className="flex items-start space-x-4">
            <div className="bg-yellow-400 p-3 rounded-lg">
              <Shield className="w-7 h-7 text-white" />
            </div>
            <div className="flex-1">
              <h3 className="text-xl font-bold text-yellow-900 mb-4">Safety Features Active</h3>
              <div className="grid grid-cols-1 md:grid-cols-2 gap-3">
                <div className="flex items-center space-x-2 text-yellow-800">
                  <CheckCircle className="w-5 h-5 text-yellow-600" />
                  <span>Maximum trade size: ${config.maxTradeSize}</span>
                </div>
                <div className="flex items-center space-x-2 text-yellow-800">
                  <CheckCircle className="w-5 h-5 text-yellow-600" />
                  <span>Daily loss limit: {config.dailyLossLimit}%</span>
                </div>
                <div className="flex items-center space-x-2 text-yellow-800">
                  <CheckCircle className="w-5 h-5 text-yellow-600" />
                  <span>API keys encrypted and secure</span>
                </div>
                <div className="flex items-center space-x-2 text-yellow-800">
                  <CheckCircle className="w-5 h-5 text-yellow-600" />
                  <span>Emergency stop available 24/7</span>
                </div>
                <div className="flex items-center space-x-2 text-yellow-800">
                  <CheckCircle className="w-5 h-5 text-yellow-600" />
                  <span>Complete audit trail logging</span>
                </div>
                <div className="flex items-center space-x-2 text-yellow-800">
                  <CheckCircle className="w-5 h-5 text-yellow-600" />
                  <span>Persistent storage backup</span>
                </div>
              </div>
            </div>
          </div>
        </div>
      </div>

      {/* Footer */}
      <footer className="bg-gradient-to-r from-gray-800 to-gray-900 text-white py-8 mt-16">
        <div className="container mx-auto px-6">
          <div className="text-center space-y-3">
            <p className="text-lg font-medium flex items-center justify-center space-x-2">
              <AlertTriangle className="w-5 h-5 text-yellow-400" />
              <span>Trading involves substantial risk of loss</span>
            </p>
            <p className="text-sm text-gray-400">
              Never trade with money you can't afford to lose. This is educational software, not financial advice.
            </p>
            <p className="text-xs text-gray-500 mt-4">
              Binance Infinite Loop Bot v2.0 • Production Ready • Built for Vercel
            </p>
          </div>
        </div>
      </footer>
    </div>
  );
};

export default App;
